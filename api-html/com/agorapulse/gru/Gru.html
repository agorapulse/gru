018 package com.agorapulse.gru;
019 
020 import com.agorapulse.gru.minions.Command;
021 import com.agorapulse.gru.minions.HttpMinion;
022 import com.agorapulse.gru.minions.Minion;
023 
024 import java.io.Closeable;
025 import java.util.ArrayList;
026 import java.util.List;
027 import java.util.function.Consumer;
028 
029 /**
030  * Gru steals the controller's unit test to verify controller's actions in context.
031  * <p>
032  */
033 public class Gru implements Closeable {
034 
035     /**
036      * Steals the unit test for himself.
037      * <p>
038      * Typical usage is <code>Gru gru = Gru.create(this)</code>
039      *
040      * @param client unit test being stolen
041      * @return new Gru instance stealing current unit test
042      * @deprecated use {@link #create(Client)} instead
043      */
044     @Deprecated
045     public static Gru equip(Client client) {
046         return create(client);
047     }
048 
049     /**
050      * Creates Gru's instance for a given unit test.
051      * <p>
052      * Typical usage is <code>@AutoCleanup Gru gru = Gru.create(this)</code>
053      *
054      * @param client unit test
055      * @return new Gru instance using the current unit test
056      */
057     public static Gru create(Client client) {
058         return new Gru(client);
059     }
060 
061     private Gru(Client client) {
062         this.client = client;
063     }
064 
065     /**
066      * Prepare every test with following configuration.
067      *
068      * @param configuration configuration applied to every feature method
069      * @return self
070      */
071     public final Gru prepare(Consumer<TestDefinitionBuilder> configuration
072     ) {
073         this.configurations.add(configuration);
074         return this;
075     }
076 
077 
078     /**
079      * Prepare tests with given base URI.
080      *
081      * @param baseUri the base URI for the test calls
082      * @return self
083      */
084     public final Gru prepare(String baseUri) {
085         this.configurations.add(c -> c.baseUri(baseUri));
086         return this;
087     }
088 
089     /**
090      * Adds minion to the squad for current test.
091      *
092      * @param minion minion to be added to the squad
093      * @return self
094      */
095     public final Gru engage(Minion minion) {
096         this.squad.add(minion);
097         return this;
098     }
099 
100 
101     /**
102      * Checks if the expectations has been verified and resets the internal state.
103      */
104     @Override
105     public void close() {
106         if (!verified) {
107             context.throwErrorIfPresent();
108             if (definition) {
109                 throw new AssertionError("Test wasn't verified. Call assertion.verify() from the then block manually!");
110             }
111         }
112 
113         reset(false);
114     }
115 
116     /**
117      * Defines and verifies API test and runs the controller initialization and the action under test.
118      * <p>
119      *
120      * @param expectation test definition
121      */
122     public final void verify(Consumer<TestDefinitionBuilder> expectation) throws Throwable {
123         try (Gru self = test(expectation)) {
124             self.verify();
125         }
126     }
127 
128     /**
129      * Defines API test and runs the controller initialization and the action under test.
130      * <p>
131      * Use this method either in when or expect block.
132      *
133      * @param expectation test definition
134      * @return self, note that when Groovy Truth is evaluated, <code>verify</code> method is called automatically
135      */
136     public final Gru test(Consumer<TestDefinitionBuilder> expectation) {
137         definition = true;
138 
139         DefaultTestDefinitionBuilder builder = new DefaultTestDefinitionBuilder(client, this.squad);
140 
141         for (Minion minion : client.getInitialSquad()) {
142             squad.add(minion);
143         }
144 
145         squad.command(HttpMinion.class, Command.noop());
146 
147         for (Consumer<TestDefinitionBuilder> configuration : configurations) {
148             configuration.accept(builder);
149         }
150 
151         expectation.accept(builder);
152 
153         checkExpectationsPresent();
154 
155         context = squad.beforeRun(client, context);
156 
157         if (!context.hasError()) {
158             try {
159                 context = client.run(squad, context);
160             } catch (Exception e) {
161                 context = context.withError(e);
162             }
163         }
164 
165         context = squad.afterRun(client, context);
166 
167         return this;
168     }
169 
170     /**
171      * Verifies all expectations.
172      *
173      * @return true if all verifications are successful
174      * @throws AssertionError if any verification fails
175      */
176     public final boolean verify() throws Throwable {
177         checkExpectationsPresent();
178 
179         if (verified) {
180             return verificationResult;
181         }
182 
183         verified = true;
184 
185         context.throwErrorIfPresent();
186 
187         squad.verify(client, context);
188 
189         return verificationResult = true;
190     }
191 
192     private void checkExpectationsPresent() {
193         if (!definition) {
194             throw new AssertionError("There are no expectations!");
195         }
196     }
197 
198     /**
199      * Reset the internal state. This is done by the rule automatically.
200      */
201     public Gru reset() {
202         return reset(true);
203     }
204 
205     /**
206      * Reset the internal state. This is done by the rule automatically.
207      *
208      * @param resetConfigurations also clear the configurations created using {@link #prepare(Consumer)} method
209      */
210     public Gru reset(boolean resetConfigurations) {
211         verified = false;
212         verificationResult = false;
213         definition = false;
214         context = GruContext.EMPTY;
215         squad = new Squad();
216         client.reset();
217 
218         if (resetConfigurations) {
219             configurations.clear();
220         }
221 
222         return this;
223     }
224 
225     private final Client client;
226     /**
227      * Additional configurations to be applied to every feature method.
228      */
229     private final List<Consumer<TestDefinitionBuilder>> configurations = new ArrayList<>();
230     /**
231      * Squad for current feature method.
232      */
233     private Squad squad = new Squad();
234     /**
235      * Context for current feature method.
236      */
237     private GruContext context = GruContext.EMPTY;
238     /**
239      * If the expectations has been already verified.
240      */
241     private boolean verified;
242 
243     /**
244      * Verification result.
245      */
246     private boolean verificationResult;
247 
248     /**
249      * IF test method has been called
250      */
251     private boolean definition;
252 }
