018 package com.agorapulse.gru;
019 
020 import com.agorapulse.gru.minions.Command;
021 import com.agorapulse.gru.minions.Minion;
022 import groovy.lang.Closure;
023 import groovy.lang.DelegatesTo;
024 import groovy.transform.stc.ClosureParams;
025 import groovy.transform.stc.FromString;
026 import groovy.transform.stc.SimpleType;
027 import net.javacrumbs.jsonunit.core.internal.JsonUtils;
028 import space.jasan.support.groovy.closure.ConsumerWithDelegate;
029 
030 import java.util.Collections;
031 import java.util.List;
032 import java.util.Map;
033 import java.util.function.Consumer;
034 
035 /**
036  * Prepares the request for controller action.
037  */
038 public interface RequestDefinitionBuilder extends WithContentSupport {
039 
040     /**
041      * @see Squad#command(Class, Closure)
042      */
043     default <M extends Minion> RequestDefinitionBuilder command(
044         @DelegatesTo.Target Class<M> minionType,
045         @DelegatesTo(genericTypeIndex = 0, strategy = Closure.DELEGATE_FIRST)
046         @ClosureParams(value = FromString.class, options = "M")
047             Closure command
048     ) {
049         return command(minionType, Command.create(command));
050     }
051 
052     /**
053      * @see Squad#command(Class, Closure)
054      */
055     <M extends Minion> RequestDefinitionBuilder command(Class<M> minionType, Command<M> command);
056 
057 
058 
059     /**
060      * Sets a JSON request from given file.
061      * The file must reside in same package as the test in the directory with the same name as the test
062      * e.g. src/test/resources/org/example/foo/MySpec.
063      * The file is created automatically during first run if it does not exist yet with empty object definition.
064      *
065      * It automatically applies Content-Type: application/json header.
066      *
067      * @param relativePath a JSON request file
068      * @return self
069      */
070     RequestDefinitionBuilder json(String relativePath);
071 
072     /**
073      * Sets a JSON request from given content.
074      *
075      * It automatically applies Content-Type: application/json header.
076      *
077      * @param content a JSON request file
078      * @return self
079      */
080     RequestDefinitionBuilder json(Content content);
081 
082     /**
083      * Sets a JSON request from given content.
084      *
085      * It automatically applies Content-Type: application/json header.
086      *
087      * @param map map to be converted to JSON
088      * @return self
089      */
090     default RequestDefinitionBuilder json(Map map) {
091         return json(inline(JsonUtils.convertToJson(map, "request").toString()));
092     }
093 
094     /**
095      * Sets a JSON request from given content.
096      *
097      * It automatically applies Content-Type: application/json header.
098      *
099      * @param list list to be converted to JSON
100      * @return self
101      */
102     default RequestDefinitionBuilder json(List list) {
103         return json(inline(JsonUtils.convertToJson(list, "request").toString()));
104     }
105 
106     /**
107      * Sets a payload for the request from given file.
108      * The file must reside in same package as the test in the directory with the same name as the test
109      * e.g. src/test/resources/org/example/foo/MySpec.
110      * The file is created automatically during first run if it does not exist yet with empty object definition.
111      *
112      * It automatically applies Content-Type header.
113      *
114      * @param relativePath relative path to the payload file for the request
115      * @param contentType content type of the payload
116      * @return self
117      */
118     RequestDefinitionBuilder content(String relativePath, String contentType);
119 
120     /**
121      * Sets a request payload from given content.
122      *
123      * It automatically applies Content-Type header.
124      *
125      * @param content payload for the request
126      * @param contentType content type of the payload
127      * @return self
128      */
129     RequestDefinitionBuilder content(Content content, String contentType);
130 
131     /**
132      * Adds URL parameters for the action execution.
133      *
134      * @param params additional URL parameters for the action execution
135      * @return self
136      */
137     RequestDefinitionBuilder params(Map<String, Object> params);
138 
139     default RequestDefinitionBuilder param(String name, String value) {
140         return params(Collections.singletonMap(name, value));
141     }
142 
143     /**
144      * Adds HTTP headers for the action execution.
145      *
146      * @param headers additional HTTP headers for the action execution
147      * @return self
148      */
149     RequestDefinitionBuilder headers(Map<String, String> headers);
150 
151     RequestDefinitionBuilder upload(Consumer<MultipartDefinitionBuilder> definition);
152 
153     default RequestDefinitionBuilder upload(
154         @DelegatesTo(value = MultipartDefinitionBuilder.class, strategy = Closure.DELEGATE_FIRST)
155         @ClosureParams(value = SimpleType.class, options = "com.agorapulse.gru.MultipartDefinitionBuilder")
156             Closure<MultipartDefinitionBuilder> definition
157     ) {
158         return upload(ConsumerWithDelegate.create(definition));
159     }
160 
161     /**
162      * Add a HTTP cookie to the action execution.
163      * @param name name of the cookie
164      * @param value value of the cookie
165      * @return self
166      */
167     default RequestDefinitionBuilder cookie(String name, String value) {
168         return cookies(Collections.singletonMap(name, value));
169     }
170 
171 
172     /**
173      * Add a HTTP cookies to the action execution.
174      * @param cookies cookies to be added
175      * @return self
176      */
177     RequestDefinitionBuilder cookies(Map<String, String> cookies);
178 }
