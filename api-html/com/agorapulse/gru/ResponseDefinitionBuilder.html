038 /**
039  * Sets expectations for the response after the controller action has been executed.
040  *
041  */
042 public interface ResponseDefinitionBuilder extends HttpStatusShortcuts, JsonUnitOptionsShortcuts, WithContentSupport {
043 
044     /**
045      * @see Squad#command(Class, Closure)
046      */
047     default <M extends Minion> ResponseDefinitionBuilder command(
048         Class<M> minionType,
049         @DelegatesTo(type = "M", strategy = Closure.DELEGATE_FIRST)
050         @ClosureParams(value = FromString.class, options = "M")
051             Closure command
052     ) {
053         return command(minionType, Command.create(command));
054     }
055 
056     /**
057      * @see Squad#command(Class, Closure)
058      */
059     <M extends Minion> ResponseDefinitionBuilder command(Class<M> minionType, Command<M> command);
060 
061     /**
062      * Sets a expected status returned.
063      * Defaults to OK.
064      *
065      * @param aStatus a expected status returned
066      * @return self
067      */
068     ResponseDefinitionBuilder status(int aStatus);
069 
070     /**
071      * Sets the acceptable statuses returned.
072      * Defaults to OK.
073      *
074      * @param statuses the acceptable statuses
075      * @return self
076      */
077     ResponseDefinitionBuilder statuses(int... statuses);
078 
079 
080     /**
081      * Sets an expected JSON response from given file.
082      * The file must reside in same package as the test in the directory with the same name as the test
083      * e.g. src/test/resources/org/example/foo/MySpec.
084      * The file is created automatically during first run if it does not exist yet with the returned JSON.
085      *
086      * @param relativePath an expected JSON response file
087      * @return self
088      */
089     ResponseDefinitionBuilder json(String relativePath);
090 
091     /**
092      * Sets an expected JSON response from given content.
093      *
094      * @param content direct content
095      * @return self
096      */
097     ResponseDefinitionBuilder json(Content content);
098 
099     /**
100      * Sets a JSON request from given content.
101      *
102      * It automatically applies Content-Type: application/json header.
103      *
104      * @param map map to be converted to JSON
105      * @return self
106      */
107     default ResponseDefinitionBuilder json(Map map) {
108         return json(inline(JsonUtils.convertToJson(map, "response").toString()));
109     }
110 
111     /**
112      * Sets a JSON request from given content.
113      *
114      * It automatically applies Content-Type: application/json header.
115      *
116      * @param list list to be converted to JSON
117      * @return self
118      */
119     default ResponseDefinitionBuilder json(List list) {
120         return json(inline(JsonUtils.convertToJson(list, "response").toString()));
121     }
122 
123 
124     /**
125      * Sets an expected HTML response from given file.
126      * The file must reside in same package as the test in the directory with the same name as the test
127      * e.g. src/test/resources/org/example/foo/MySpec.
128      * The file is created automatically during first run if it does not exist yet with the returned HTML.
129      *
130      * @param relativePath an expected HTML response file
131      * @return self
132      */
133     ResponseDefinitionBuilder html(String relativePath);
134 
135     /**
136      * Sets an expected HTML response from given content.
137      *
138      *  @param content an expected HTML response file
139      * @return self
140      */
141     ResponseDefinitionBuilder html(Content content);
142 
143     /**
144      * Sets an expected text response from given file.
145      * The file must reside in same package as the test in the directory with the same name as the test
146      * e.g. src/test/resources/org/example/foo/MySpec.
147      * The file is created automatically during first run if it does not exist yet with the returned HTML.
148      *
149      * @param relativePath an expected HTML response file
150      * @return self
151      */
152     ResponseDefinitionBuilder text(String relativePath);
153     /**
154      * Sets an expected text response from given content.
155      *
156      * @param content an expected HTML response file
157      * @return self
158      */
159     ResponseDefinitionBuilder text(Content content);
160 
161     /**
162      * Sets additional assertions and configuration for JsonFluentAssert instance which is testing the response.
163      *
164      * @param additionalConfiguration additional assertions and configuration for JsonFluentAssert instance
165      * @return self
166      */
167     default ResponseDefinitionBuilder json(
168         @DelegatesTo(value = JsonFluentAssert.ConfigurableJsonFluentAssert.class, strategy = Closure.DELEGATE_FIRST)
169         @ClosureParams(value = SimpleType.class, options = "net.javacrumbs.jsonunit.fluent.JsonFluentAssert.ConfigurableJsonFluentAssert")
170             Closure<JsonFluentAssert.ConfigurableJsonFluentAssert> additionalConfiguration
171     ) {
172         return json(FunctionWithDelegate.create(additionalConfiguration));
173     }
174 
175     /**
176      * Sets additional assertions and configuration for JsonFluentAssert instance which is testing the response.
177      *
178      * @param additionalConfiguration additional assertions and configuration for JsonFluentAssert instance
179      * @return self
180      */
181     DefaultResponseDefinitionBuilder json(Function<JsonFluentAssert.ConfigurableJsonFluentAssert, JsonFluentAssert.ConfigurableJsonFluentAssert> additionalConfiguration);
182 
183     /**
184      * Adds HTTP headers which are expected to be returned after action execution.
185      *
186      * @param additionalHeaders additional HTTP headers which are expected to be returned after action execution
187      * @return self
188      */
189     ResponseDefinitionBuilder headers(Map<String, String> additionalHeaders);
190 
191     /**
192      * Sets the expected redirection URI.
193      *
194      * @param uri expected URI
195      * @return self
196      */
197     ResponseDefinitionBuilder redirect(String uri);
198 
199     /**
200      * Sets an expected JSON response from given file.
201      * The file must reside in same package as the test in the directory with the same name as the test
202      * e.g. src/test/resources/org/example/foo/MySpec.
203      * The file is created automatically during first run if it does not exist yet with the returned JSON.
204      *
205      * @param relativePath an expected JSON response file
206      * @param option       JsonUnit option, e.g. IGNORING_EXTRA_ARRAY_ITEMS
207      * @return self other JsonUnit options e.g. IGNORING_EXTRA_ARRAY_ITEMS
208      */
209     ResponseDefinitionBuilder json(String relativePath, Option option, Option... options);
210 
211     /**
212      * Expect a cookie returned by the server.
213      * @param cookieDefinition definition of the cookie
214      * @return self
215      */
216     default ResponseDefinitionBuilder cookie(
217         @DelegatesTo(value = ResponseCookieDefinition.class, strategy = Closure.DELEGATE_FIRST)
218         @ClosureParams(value = SimpleType.class, options = "com.agorapulse.gru.ResponseCookieDefinition")
219             Closure<ResponseCookieDefinition> cookieDefinition
220     ) {
221         return cookie(ConsumerWithDelegate.create(cookieDefinition));
222     }
223 
224     /**
225      * Expect a cookie returned by the server.
226      * @param cookieDefinition definition of the cookie
227      * @return self
228      */
229     ResponseDefinitionBuilder cookie(Consumer<ResponseCookieDefinition> cookieDefinition);
230 
231     default ResponseDefinitionBuilder cookies(Map<String, String> cookies) {
232         cookies.forEach((name, value) -> cookie((c) -> c.name(name).value(value)));
233         return this;
234     }
