058         @SuppressWarnings("unchecked")
059         String executeHandler(ApiGatewayProxyRequest request, Object unitTest) throws ClassNotFoundException, IOException, IllegalAccessException, InvocationTargetException, InstantiationException {
060             String[] parts = handler.split("::");
061             String className = parts[0];
062             String methodName = parts[1];
063 
064             Class clazz = Class.forName(className);
065 
066             if (RequestStreamHandler.class.isAssignableFrom(clazz)) {
067                 return handleRequestStreamHandler(request, clazz);
068             }
069 
070             Method method = Arrays.stream(clazz.getMethods())
071                 .filter(m -> m.getGenericParameterTypes().length == 2 && Context.class.equals(m.getGenericParameterTypes()[1]))
072                 .min(comparing((Method m) -> distanceToObject(m.getParameterTypes()[0])).reversed())
073                 .orElseThrow(() -> new IllegalArgumentException("Cannot find method " + methodName + " in " + className + " matching the signature contract"));
074 
075             Type firstParameterType = method.getGenericParameterTypes()[0];
076 
077             Class firstParameter = null;
078 
079             if (firstParameterType instanceof Class) {
080                 firstParameter = (Class) firstParameterType;
081             } else if (firstParameterType instanceof TypeVariable) {
082                 TypeVariable variable = (TypeVariable) firstParameterType;
083                 int indexOfTypeVariable = Arrays.asList(clazz.getSuperclass().getTypeParameters()).indexOf(variable);
084                 if (indexOfTypeVariable >= 0) {
085                     Type genericSuperclass = clazz.getGenericSuperclass();
086                     if (genericSuperclass instanceof ParameterizedType) {
087                         Type actualTypeArgument = ((ParameterizedType) genericSuperclass).getActualTypeArguments()[indexOfTypeVariable];
088                         if (actualTypeArgument instanceof Class) {
089                             firstParameter = (Class) actualTypeArgument;
090                         }
091                     }
092                 }
093             }
094 
095             if (firstParameter == null) {
096                 throw new IllegalArgumentException("Cannot determine type of the first parameter of method " + methodName);
097             }
098 
099             ObjectMapper mapper = new ObjectMapper();
100 
101             // TODO: handle streaming handler
102 
103             request.setPathParameters(route.extractPathParameters(request.getPath()));
104 
105             Object input = mapper.readValue(prepareRequestObject(request), firstParameter);
106 
107             Object result = method.invoke(getOrCreateHandler(unitTest, clazz), input, request.getContext());
108 
109             OutputStream outputStream = new ByteArrayOutputStream();
110             mapper.writeValue(outputStream, prepareResponseObject(result));
111             return outputStream.toString();
112         }

114         private String handleRequestStreamHandler(ApiGatewayProxyRequest request, Class<? extends RequestStreamHandler> handlerClass) throws IllegalAccessException, InstantiationException, IOException {
115             RequestStreamHandler streamHandler = handlerClass.newInstance();
116 
117             ByteArrayInputStream bais = new ByteArrayInputStream(prepareRequestObject(request).getBytes());
118             ByteArrayOutputStream baos = new ByteArrayOutputStream();
119 
120             streamHandler.handleRequest(bais, baos, request.getContext());
121 
122             baos.flush();
123 
124             return baos.toString();
125         }

149         private int distanceToObject(Class<?> clazz, int initial) {
150             if (Object.class.equals(clazz) || clazz == null) {
151                 return initial;
152             }
153             return distanceToObject(clazz.getSuperclass(), initial + 1);
154         }

156         private <T> T getOrCreateHandler(Object unitTest, Class<T> clazz) {
157             return Arrays.stream(unitTest.getClass().getDeclaredFields())
158                 .filter(f -> clazz.equals(f.getType()))
159                 .findFirst()
160                 .map(f -> getFieldValue(unitTest, clazz, f))
161                 .orElseGet(() -> createNewInstance(clazz));
162         }

310         boolean matches(String method, String url) {
311             return methods.contains(method) && this.url.matcher(url).matches();
312         }
313 
314         Map<String, String> extractPathParameters(String path) {
315             Matcher matcher = this.url.matcher(path);
316 
317             if (!matcher.matches()) {
318                 throw new IllegalStateException("Path " + path + " does not match!");
319             }
320 
321             Map<String, String> ret = new LinkedHashMap<>();
322 
323             for (String pathParameter : pathParametersNames) {
324                 ret.put(pathParameter, matcher.group(pathParameter));
325             }
326 
327             return ret;
328         }

330         private Pattern createPatternForUrlMapping(String url) {
331             if (url.contains("{")) {
332                 return Pattern.compile(url.replaceAll("\\{([^\\}]+)\\}", "(?<$1>.*?)"));
333             }
334             return Pattern.compile(Pattern.quote(url));
335         }
336 
337         private Set<String> createNamedParametersNames(String url) {
338             if (url.contains("{")) {
339                 Set<String> ret = new LinkedHashSet<>();
340                 Matcher matcher = Pattern.compile("\\{([^\\}]+)\\}").matcher(url);
341                 while (matcher.find()) {
342                     ret.add(matcher.group(1));
343                 }
344                 return ret;
345             }
346             return Collections.emptySet();
347         }

349         private Mapping to(String handler, Consumer<MappingConfiguration> configuration, boolean proxied) {
350             Mapping mapping = new Mapping(this, handler.contains("::") ? handler : (handler + "::handleRequest"), proxied);
351             Optional.ofNullable(configuration).ifPresent(mapping::configure);
352             self.mappings.add(mapping);
353             return mapping;
354         }

357     private final List<Mapping> mappings = new ArrayList<>();

359     public Route map(String url, String... httpMethods) {
360         Collection<String> methods;
361 
362         if (httpMethods.length == 0) {
363             methods = Arrays.asList(HEAD, POST, PUT, PATCH, DELETE, OPTIONS, TRACE, GET);
364         } else {
365             methods = Arrays.asList(httpMethods);
366         }
367 
368         return new Route(this, url, methods);
369     }

371     Mapping findMapping(String method, String url) {
372         for (Mapping mapping : mappings) {
373             if (mapping.route.matches(method, url)) {
374                 return mapping;
375             }
376         }
377         throw new IllegalArgumentException(String.format("Route %s: %s is not mapped!", method, url));
378     }
