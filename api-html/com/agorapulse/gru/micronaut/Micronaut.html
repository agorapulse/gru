049     /**

052     public static class MicronautApplicationBuilder {
053         private final Object unitTest;
054         private Consumer<ApplicationContextBuilder> contextBuilderConfiguration = b -> {};
055         private Consumer<ApplicationContext> contextConfiguration = b -> {};
056 
057         MicronautApplicationBuilder(Object unitTest) {
058             this.unitTest = unitTest;
059         }
060 
061         /**
062          * Customize the application context.
063          * @param contextConfiguration application context configuration
064          * @return self
065          */
066         public MicronautApplicationBuilder doWithContext(Consumer<ApplicationContext> contextConfiguration) {
067             this.contextConfiguration = this.contextConfiguration.andThen(contextConfiguration);
068             return this;
069         }
070

072         /**
073          * Customize the application context builder.
074          * @param contextBuilderConfiguration application context configuration
075          * @return self
076          */
077         public MicronautApplicationBuilder doWithContextBuilder(Consumer<ApplicationContextBuilder> contextBuilderConfiguration) {
078             this.contextBuilderConfiguration = this.contextBuilderConfiguration.andThen(contextBuilderConfiguration);
079             return this;
080         }
081 
082         /**
083          * Starts the internal {@link ApplicationContext} and returns the client instance.
084          * @return the new client instance
085          */
086         public Client start() {
087             SelfStartingContextProvider provider = new SelfStartingContextProvider(this);
088             provider.getApplicationContext().inject(unitTest);
089             return create(unitTest, provider);
090         }
091     }

093     private static class SelfStartingContextProvider implements ApplicationContextProvider, Closeable {
094 
095         private final MicronautApplicationBuilder micronautApplicationBuilder;
096         private ApplicationContext context;
097 
098         private SelfStartingContextProvider(MicronautApplicationBuilder builder) {
099             this.micronautApplicationBuilder = builder;
100         }
101 
102         @Override
103         public ApplicationContext getApplicationContext() {
104             if (context != null) {
105                 return context;
106             }
107 
108             ApplicationContextBuilder builder = ApplicationContext.builder();
109             micronautApplicationBuilder.contextBuilderConfiguration.accept(builder);
110 
111             context = builder.build();
112             micronautApplicationBuilder.contextConfiguration.accept(context);
113 
114             context.start();
115 
116             EmbeddedServer application = context.getBean(EmbeddedServer.class);
117 
118             application.start();
119 
120             return context;
121         }
122 
123         @Override
124         public void close() {
125             context.close();
126         }
127     }

145      *

237         if (contextProvider instanceof Closeable) {

279     private static void assertProviderIfPossible(Object unitTest) {
280         if (unitTest == null || unitTest instanceof ApplicationContextProvider) {
281             return;
282         }
283 
284         if (Arrays.stream(unitTest.getClass().getAnnotations()).anyMatch(a -> a.getClass().getSimpleName().equals("MicronautTest"))) {
285             throw new IllegalArgumentException(
286                 "Specification "
287                     + unitTest.getClass().getName()
288                     + " must either implement "
289                     + ApplicationContextProvider.class.getName()
290                     + " or use inject the instance using @Inject Gru gru"
291             );
292         }
293 
294         Optional<Field> contextField = Arrays.stream(unitTest.getClass().getFields()).filter(f -> ApplicationContext.class.isAssignableFrom(f.getType())).findFirst();
295         if (contextField.isPresent()) {
296             throw new IllegalArgumentException(
297                 "Specification "
298                     + unitTest.getClass().getName()
299                     + " must either implement "
300                     + ApplicationContextProvider.class.getName()
301                     + " returning value of field "
302                     + contextField.get().getName()
303                     + " or use Micronaut.create(this) { "
304                     + contextField.get().getName()
305                     + " } instead of Micronaut.create(this)"
306             );
307         }
308     }
