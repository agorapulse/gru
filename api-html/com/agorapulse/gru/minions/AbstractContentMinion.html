018 package com.agorapulse.gru.minions;
019 
020 import com.agorapulse.gru.Client;
021 import com.agorapulse.gru.Content;
022 import com.agorapulse.gru.GruContext;
023 import com.agorapulse.gru.Squad;
024 import com.agorapulse.gru.content.ContentUtils;
025 
026 import java.io.ByteArrayInputStream;
027 import java.io.InputStream;
028 import java.util.ArrayList;
029 import java.util.List;
030 import java.util.stream.Collectors;
031 
032 /**
033  * Minion responsible for JSON requests and responses.
034  */
035 public abstract class AbstractContentMinion<C extends Client> extends AbstractMinion<C> {
036 
037 
038     public static final String TEST_RESOURCES_FOLDER_PROPERTY_NAME = "TEST_RESOURCES_FOLDER";
039     public static final String ENVIRONMENT_VARIABLE_REWRITE_RESOURCES = "COM_AGORAPULSE_GRU_REWRITE";
040     public static final String SYSTEM_PROPERTY_REWRITE_RESOURCES = "com.agorapulse.gru.rewrite";
041     private Content responseContent;
042     private final boolean rewrite;
043     protected final List<Content> createdResources = new ArrayList<>();
044 
045     protected AbstractContentMinion(Class<C> clientType) {
046         super(clientType);
047         rewrite = Boolean.TRUE.toString().equals(System.getenv(ENVIRONMENT_VARIABLE_REWRITE_RESOURCES)) || Boolean.TRUE.toString().equals(System.getProperty(SYSTEM_PROPERTY_REWRITE_RESOURCES));
048     }
049 
050     @Override
051     public final void doVerify(Client client, Squad squad, GruContext context) throws Throwable {
052         if (responseContent != null) {
053             final InputStream is = responseContent.load(client);
054             String expectedResponseText = is == null ? null : ContentUtils.getText(is);
055             String responseText = readResponseText(client, squad, context);
056 
057             if (expectedResponseText == null && responseText != null) {
058                 expectedResponseText = responseText;
059                 if (responseContent.isSaveSupported()) {
060                     responseContent.save(client, new ByteArrayInputStream(normalize(expectedResponseText).getBytes()));
061                     System.err.println("Content is missing for " + getResponseContent() + ". New file was created with content:\n" + expectedResponseText);
062                     createdResources.add(responseContent);
063                 }
064             }
065 
066             if (expectedResponseText != null) {
067                 try {
068                     similar(normalize(responseText), normalize(expectedResponseText));
069                 } catch (AssertionError assertionError) {
070                     if (rewrite && isRewriteSupported() && responseContent.isSaveSupported()) {
071                         rewrite(client, normalize(responseText), normalize(expectedResponseText));
072                     } else {
073                         throw assertionError;
074                     }
075                 }
076             }
077         }
078 
079 
080         if (!createdResources.isEmpty()) {
081             throw new AssertionError("Fixture files were created or updated: " + createdResources.stream().map(Object::toString).collect(Collectors.joining(", ")) + ". Please, run the test again to verify it is repeatable.");
082         }
083 
084     }
085 
086     public Content getResponseContent() {
087         return responseContent;
088     }
089 
090     public void setResponseContent(Content responseContent) {
091         this.responseContent = responseContent;
092     }
093 
094     protected abstract void similar(String actual, String expected) throws AssertionError;
095 
096     protected String normalize(String input) {
097         return input;
098     }
099 
100     protected boolean isRewriteSupported() {
101         return false;
102     }
103 
104     protected String rewrite(Client client, String actualText, String expectedText) {
105         throw new UnsupportedOperationException("Rewrite is not supported");
106     }
107 
108     protected String readResponseText(Client client, Squad squad, GruContext context) {
109         return client.getResponse().getText();
110     }
111 
112 }
