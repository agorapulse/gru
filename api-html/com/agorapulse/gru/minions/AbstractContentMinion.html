18 package com.agorapulse.gru.minions
19 
20 import com.agorapulse.gru.Client
21 import com.agorapulse.gru.Content
22 import com.agorapulse.gru.GruContext
23 import com.agorapulse.gru.Squad
24 import groovy.transform.CompileStatic
25 import groovy.util.logging.Log
26 
27 /**
28  * Minion responsible for JSON requests and responses.
29  */
30 @Log @CompileStatic
31 abstract class AbstractContentMinion<C extends Client> extends AbstractMinion<C> {
32 
33     public static final String TEST_RESOURCES_FOLDER_PROPERTY_NAME = 'TEST_RESOURCES_FOLDER'
34     public static final String ENVIRONMENT_VARIABLE_REWRITE_RESOURCES = 'COM_AGORAPULSE_GRU_REWRITE'
35     public static final String SYSTEM_PROPERTY_REWRITE_RESOURCES = 'com.agorapulse.gru.rewrite'
36 
37     Content responseContent
38     boolean rewrite
39 
40     protected final List<String> createdResources = []
41 
42     protected AbstractContentMinion(Class<C> clientType) {
43         super(clientType)
44         rewrite = System.getenv(ENVIRONMENT_VARIABLE_REWRITE_RESOURCES) == true.toString() ||
45             System.getProperty(SYSTEM_PROPERTY_REWRITE_RESOURCES) == true.toString()
46     }
47 
48     @Override
49     final void doVerify(Client client, Squad squad, GruContext context) throws Throwable {
50         if (responseContent) {
51             String expectedResponseText = responseContent?.load(client)?.text
52             String responseText = readResponseText(client, squad, context)
53 
54             if (!expectedResponseText && responseText) {
55                 expectedResponseText = responseText
56                 if (responseContent.saveSupported) {
57                     responseContent.save(client, new ByteArrayInputStream(normalize(expectedResponseText).bytes))
58                     log.warning("Content is missing for $responseContent. New file was created with content:\n$expectedResponseText")
59                     createdResources << responseContent.toString()
60                 }
61             }
62 
63             if (expectedResponseText) {
64                 try {
65                     similar(normalize(responseText), normalize(expectedResponseText))
66                 } catch (AssertionError assertionError) {
67                     if (rewrite && rewriteSupported && responseContent.saveSupported) {
68                         rewrite(client, normalize(responseText), normalize(expectedResponseText))
69                     } else {
70                         throw assertionError
71                     }
72                 }
73             }
74         }
75 
76         if (createdResources) {
77             throw new AssertionError("Fixture files were created or updated: ${createdResources.join(', ')}. " +
78                 'Please, run the test again to verify it is repeatable.')
79         }
80     }
81 
82     protected abstract void similar(String actual, String expected) throws AssertionError
83 
84     protected String normalize(String input) { input }
85 
86     protected boolean isRewriteSupported() { false }
87 
88     @SuppressWarnings(['ConfusingMethodName', 'UnusedMethodParameter'])
89     protected String rewrite(Client client, String actualText, String expectedText) {
90         throw new UnsupportedOperationException('Rewrite is not supported')
91     }
92 
93     @SuppressWarnings('UnusedMethodParameter')
94     protected String readResponseText(Client client, Squad squad, GruContext context) { client.response.text }
95 }
