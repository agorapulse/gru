18 package com.agorapulse.gru.minions

20 import com.agorapulse.gru.Client
21 import groovy.transform.CompileStatic
22 import groovy.util.logging.Log
23 import org.jsoup.Jsoup
24 import org.jsoup.nodes.Document
25 import org.jsoup.nodes.DocumentType
26 import org.jsoup.nodes.Entities
27 import org.jsoup.nodes.Node
28 import org.xmlunit.builder.DiffBuilder
29 import org.xmlunit.builder.Input
30 import org.xmlunit.diff.DifferenceEvaluator
31 import org.xmlunit.diff.DifferenceEvaluators
32 import org.xmlunit.diff.Diff
33 import org.xmlunit.diff.Comparison
34 import org.xmlunit.diff.ComparisonResult
35 
36 import javax.xml.transform.Source
37 
38 /**
39  * Minion responsible for JSON requests and responses.
40  */
41 @Log @CompileStatic
42 class HtmlMinion extends AbstractContentMinion<Client> {
43 
44     final int index = HTML_MINION_INDEX
45 
46     HtmlMinion() {
47         super(Client)
48     }
49 
50     @SuppressWarnings('Instanceof')
51     protected String normalize(String html) {
52         Document document = Jsoup.parse(html ?: '')
53         List<Node> documentNodes = []
54         document.childNodes().each { Node it ->
55             if (it instanceof DocumentType) {
56                 documentNodes.add(it as Node)
57             }
58         }
59         documentNodes.each {it.remove() }
60         document.outputSettings(new Document.OutputSettings().syntax(Document.OutputSettings.Syntax.xml).escapeMode(Entities.EscapeMode.xhtml))
61         document.outerHtml()
62     }
63 
64     @SuppressWarnings('GStringExpressionWithinString')
65     protected void similar(String actual, String expected) throws AssertionError {
66         String entitiesDeclaration = HtmlMinion.getResourceAsStream('entities.dtd').text
67         Source actualSource = Input.fromString(entitiesDeclaration + actual).build()
68         Source expectedSource = Input.fromString(entitiesDeclaration + expected).build()
69         Diff diff = DiffBuilder.compare(actualSource)
70         .ignoreComments()
71         .ignoreWhitespace()
72         .normalizeWhitespace()
73         .withTest(expectedSource)
74         .withDifferenceEvaluator(DifferenceEvaluators.chain(DifferenceEvaluators.Default, new DifferenceEvaluator() {
75             @Override
76             ComparisonResult evaluate(Comparison comparison, ComparisonResult outcome) {
77                 if (outcome == ComparisonResult.DIFFERENT) {
78                     if (comparison.testDetails.value == '${xml-unit.ignore}') {
79                         return ComparisonResult.EQUAL
80                     }
81                 }
82                 return outcome
83             }
84         }))
85         .build()
86 
87         if (diff.hasDifferences()) {
88             log.info 'ACTUAL:'
89             log.info actual
90             log.info 'EXPECTED:'
91             log.info expected
92             throw new AssertionError((Object) "HTML files are different:\n\n ${diff.toString()}")
93         }
94     }
95 }
