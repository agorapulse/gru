18 package com.agorapulse.gru.minions;
19 
20 import com.agorapulse.gru.Client;
21 import com.agorapulse.gru.GruContext;
22 import com.agorapulse.gru.Squad;
23 
24 import java.util.Comparator;
25 
26 /**
27  * Minions helps Gru with setting up and verifying the tests.
28  */
29 public interface Minion {
30 
31     // tag::constants[]
32     int INITIALIZATION_MINION_INDEX = -10000;
33     int HTTP_MINION_INDEX = 0;
34     int URL_MAPPINGS_MINION_INDEX = 10000;
35     int PARAMETERS_MINION_INDEX = 20000;
36     int COOKIES_MINION_INDEX = 25000;
37     int CONTENT_MINION_INDEX = 29999;
38     int JSON_MINION_INDEX = 30000;
39     int HTML_MINION_INDEX = 30001;
40     int TEXT_MINION_INDEX = 30002;
41     int MULTIPART_MINION_INDEX = 30003;
42     int INTERCEPTORS_MINION_INDEX = 50000;
43     int MODEL_MINION_INDEX = 60000;
44     // end::constants[]
45 
46     /**
47      * Comparator to compare miniorns by their index.
48      */
49     Comparator<Minion> COMPARATOR = new Comparator<Minion>() {
50         @Override
51         public int compare(Minion o1, Minion o2) {
52             return Integer.compare(o1.getIndex(), o2.getIndex());
53         }
54     };
55 
56     // tag::methods[]
57     /**
58      * @return index of the minion to so they they handle the test in given order
59      */
60     int getIndex();
61 
62     /**
63      * Prepares the execution of controller action.
64      * @param client controller unit test
65      * @param squad minion's colleagues
66      * @param context context of execution, in this phase you should only add errors if some conditions are not met
67      * @return new context holding any exception which happens during the preparation or the initial context from parameter
68      */
69     GruContext beforeRun(Client client, Squad squad, GruContext context);
70 
71     /**
72      * Modifies the result of the execution of controller action.
73      *
74      * Minions handle the result in reversed order.
75      *
76      * @param client controller unit test
77      * @param squad minion's colleagues
78      * @param context context of execution, in this phase can modify the result or add exception
79      * @return modified context or the initial one from the parameter
80      */
81     GruContext afterRun(Client client, Squad squad, GruContext context);
82 
83     /**
84      * Verifies the response after execution of the controller action.
85      *
86      * @param client controller unit test
87      * @param squad minion's colleagues
88      * @param context context of the execution, in this phase you can't modify the context, you can just throw any {@link Throwable} based on the current context state
89      * @throws Throwable if any of the conditions are not met
90      */
91     void verify(Client client, Squad squad, GruContext context) throws Throwable;
92     // end::methods[]
93 
94 }
