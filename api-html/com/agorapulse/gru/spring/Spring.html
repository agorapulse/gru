018 package com.agorapulse.gru.spring;
019 
020 import com.agorapulse.gru.AbstractClient;
021 import com.agorapulse.gru.GruContext;
022 import com.agorapulse.gru.Squad;
023 import org.springframework.mock.web.MockMultipartFile;
024 import org.springframework.test.web.servlet.MockMvc;
025 import org.springframework.test.web.servlet.MvcResult;
026 import org.springframework.test.web.servlet.request.MockHttpServletRequestBuilder;
027 import org.springframework.test.web.servlet.request.MockMultipartHttpServletRequestBuilder;
028 
029 import java.net.URI;
030 import java.util.Arrays;
031 import java.util.function.Consumer;
032 
033 import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.fileUpload;
034 import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.request;
035 
036 /**
037  * Gru implementation based on MockMvc.
038  */
039 public class Spring extends AbstractClient {
040 
041     @Deprecated
042     public static Spring steal(Object unitTest) {
043         return create(unitTest);
044     }
045 
046     public static Spring create(Object unitTest) {
047         return new Spring(unitTest);
048     }
049 
050     private GruSpringRequest request;
051     private GruSpringResponse response;
052 
053     private Spring(Object unitTest) {
054         super(unitTest);
055         reset();
056     }
057 
058     @Override
059     public GruSpringRequest getRequest() {
060         return request;
061     }
062 
063     @Override
064     public GruSpringResponse getResponse() {
065         return response;
066     }
067 
068     @Override
069     public void reset() {
070         request = new  GruSpringRequest();
071         response = null;
072     }
073 
074     @Override
075     public GruContext run(Squad squad, GruContext context) {
076         MockMvc mockMvc = findMockMvc();
077         MockHttpServletRequestBuilder builder = request.getMultipart() != null ? fileUpload(getRequestURI()) : request(request.getMethod(), getRequestURI());
078 
079         for (Consumer<MockHttpServletRequestBuilder> step : request.getSteps()) {
080             step.accept(builder);
081         }
082 
083         if (request.getMultipart() != null) {
084             MockMultipartHttpServletRequestBuilder upload = (MockMultipartHttpServletRequestBuilder) builder;
085             request.getMultipart().getParameters().forEach((k, v) -> {
086                 builder.param(k, v == null ? null :String.valueOf(v));
087             });
088             request.getMultipart().getFiles().forEach((k, f) -> {
089                 upload.file(new MockMultipartFile(
090                     f.getParameterName(),
091                     f.getFilename(),
092                     f.getContentType(),
093                     f.getBytes()
094                 ));
095             });
096         }
097 
098         try {
099             MvcResult result = mockMvc.perform(builder).andReturn();
100             response = new GruSpringResponse(result.getResponse());
101             return context.withResult(result);
102         } catch (Exception e) {
103             return context.withError(e);
104         }
105     }
106 
107     private MockMvc findMockMvc() {
108         if (getUnitTest() instanceof HasMockMvc) {
109             HasMockMvc test = (HasMockMvc) getUnitTest();
110             return test.getMockMvc();
111         }
112 
113         System.err.println("Please, let the unit test implement HasMockMvc interface to avoid unnecessary reflection.");
114 
115         return Arrays.stream(getUnitTestClass().getDeclaredFields())
116             .filter(f -> MockMvc.class.isAssignableFrom(f.getType()))
117             .findFirst()
118             .map(f -> {
119                 try {
120                     f.setAccessible(true);
121                     return (MockMvc) f.get(getUnitTest());
122                 } catch (IllegalAccessException e) {
123                     throw new IllegalStateException("Cannot read field " + f, e);
124                 }
125             })
126             .orElseThrow(() -> new IllegalStateException("MockMvc is missing in the unit test or it is null. Please provide '@Autowired MockMvc mockMvc' field in your specification"));
127     }
128 
129     private URI getRequestURI() {
130         return URI.create(((request.getBaseUri()  == null ? "" : request.getBaseUri()) + (request.getUri() == null ? "" : request.getUri())).replaceAll("/+", "/"));
131     }
