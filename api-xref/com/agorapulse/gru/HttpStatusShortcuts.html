20  /**
21   * Provides HTTP statuses' shortcuts for <code>expect</code> block so they can be used without additional imports.
22   */
23  //CHECKSTYLE:OFF
24  public interface HttpStatusShortcuts {
25  
26      /**
27       * {@code 100 Continue}.
28       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.2.1">HTTP/1.1: Semantics and Content, section 6.2.1</a>
29       */
30      int CONTINUE = 100;
31      /**
32       * {@code 101 Switching Protocols}.
33       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.2.2">HTTP/1.1: Semantics and Content, section 6.2.2</a>
34       */
35      int SWITCHING_PROTOCOLS = 101;
36      /**
37       * {@code 102 Processing}.
38       * @see <a href="http://tools.ietf.org/html/rfc2518#section-10.1">WebDAV</a>
39       */
40      int PROCESSING = 102;
41      /**
42       * {@code 103 Checkpoint}.
43       * @see <a href="http://code.google.com/p/gears/wiki/ResumableHttpRequestsProposal">A proposal for supporting
44       * resumable POST/PUT HTTP requests in HTTP/1.0</a>
45       */
46      int CHECKPOINT = 103;
47  
48      // 2xx Success
49  
50      /**
51       * {@code 200 OK}.
52       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.1">HTTP/1.1: Semantics and Content, section 6.3.1</a>
53       */
54      int OK = 200;
55      /**
56       * {@code 201 Created}.
57       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.2">HTTP/1.1: Semantics and Content, section 6.3.2</a>
58       */
59      int CREATED = 201;
60      /**
61       * {@code 202 Accepted}.
62       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.3">HTTP/1.1: Semantics and Content, section 6.3.3</a>
63       */
64      int ACCEPTED = 202;
65      /**
66       * {@code 203 Non-Authoritative Information}.
67       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.4">HTTP/1.1: Semantics and Content, section 6.3.4</a>
68       */
69      int NON_AUTHORITATIVE_INFORMATION = 203;
70      /**
71       * {@code 204 No Content}.
72       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.5">HTTP/1.1: Semantics and Content, section 6.3.5</a>
73       */
74      int NO_CONTENT = 204;
75      /**
76       * {@code 205 Reset Content}.
77       * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.3.6">HTTP/1.1: Semantics and Content, section 6.3.6</a>
78       */
79      int RESET_CONTENT = 205;
80      /**
81       * {@code 206 Partial Content}.
82       * @see <a href="http://tools.ietf.org/html/rfc7233#section-4.1">HTTP/1.1: Range Requests, section 4.1</a>
83       */
84      int PARTIAL_CONTENT = 206;
85      /**
86       * {@code 207 Multi-Status}.
87       * @see <a href="http://tools.ietf.org/html/rfc4918#section-13">WebDAV</a>
88       */
89      int MULTI_STATUS = 207;
90      /**
91       * {@code 208 Already Reported}.
92       * @see <a href="http://tools.ietf.org/html/rfc5842#section-7.1">WebDAV Binding Extensions</a>
93       */
94      int ALREADY_REPORTED = 208;
95      /**
96       * {@code 226 IM Used}.
97       * @see <a href="http://tools.ietf.org/html/rfc3229#section-10.4.1">Delta encoding in HTTP</a>
98       */
99      int IM_USED = 226;
100 
101     // 3xx Redirection
102 
103     /**
104      * {@code 300 Multiple Choices}.
105      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.1">HTTP/1.1: Semantics and Content, section 6.4.1</a>
106      */
107     int MULTIPLE_CHOICES = 300;
108     /**
109      * {@code 301 Moved Permanently}.
110      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.2">HTTP/1.1: Semantics and Content, section 6.4.2</a>
111      */
112     int MOVED_PERMANENTLY = 301;
113     /**
114      * {@code 302 Found}.
115      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.3">HTTP/1.1: Semantics and Content, section 6.4.3</a>
116      */
117     int FOUND = 302;
118     /**
119      * {@code 302 Moved Temporarily}.
120      * @see <a href="http://tools.ietf.org/html/rfc1945#section-9.3">HTTP/1.0, section 9.3</a>
121      * @deprecated in favor of {@link #FOUND} which will be returned from {@code HttpStatus.valueOf(302)}
122      */
123     @Deprecated
124     int MOVED_TEMPORARILY = 302;
125     /**
126      * {@code 303 See Other}.
127      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.4">HTTP/1.1: Semantics and Content, section 6.4.4</a>
128      */
129     int SEE_OTHER = 303;
130     /**
131      * {@code 304 Not Modified}.
132      * @see <a href="http://tools.ietf.org/html/rfc7232#section-4.1">HTTP/1.1: Conditional Requests, section 4.1</a>
133      */
134     int NOT_MODIFIED = 304;
135     /**
136      * {@code 305 Use Proxy}.
137      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.5">HTTP/1.1: Semantics and Content, section 6.4.5</a>
138      * @deprecated due to security concerns regarding in-band configuration of a proxy
139      */
140     @Deprecated
141     int USE_PROXY = 305;
142     /**
143      * {@code 307 Temporary Redirect}.
144      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.4.7">HTTP/1.1: Semantics and Content, section 6.4.7</a>
145      */
146     int TEMPORARY_REDIRECT = 307;
147     /**
148      * {@code 308 Permanent Redirect}.
149      * @see <a href="http://tools.ietf.org/html/rfc7238">RFC 7238</a>
150      */
151     int PERMANENT_REDIRECT = 308;
152 
153     // --- 4xx Client Error ---
154 
155     /**
156      * {@code 400 Bad Request}.
157      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.1">HTTP/1.1: Semantics and Content, section 6.5.1</a>
158      */
159     int BAD_REQUEST = 400;
160     /**
161      * {@code 401 Unauthorized}.
162      * @see <a href="http://tools.ietf.org/html/rfc7235#section-3.1">HTTP/1.1: Authentication, section 3.1</a>
163      */
164     int UNAUTHORIZED = 401;
165     /**
166      * {@code 402 Payment Required}.
167      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.2">HTTP/1.1: Semantics and Content, section 6.5.2</a>
168      */
169     int PAYMENT_REQUIRED = 402;
170     /**
171      * {@code 403 Forbidden}.
172      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.3">HTTP/1.1: Semantics and Content, section 6.5.3</a>
173      */
174     int FORBIDDEN = 403;
175     /**
176      * {@code 404 Not Found}.
177      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.4">HTTP/1.1: Semantics and Content, section 6.5.4</a>
178      */
179     int NOT_FOUND = 404;
180     /**
181      * {@code 405 Method Not Allowed}.
182      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.5">HTTP/1.1: Semantics and Content, section 6.5.5</a>
183      */
184     int METHOD_NOT_ALLOWED = 405;
185     /**
186      * {@code 406 Not Acceptable}.
187      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.6">HTTP/1.1: Semantics and Content, section 6.5.6</a>
188      */
189     int NOT_ACCEPTABLE = 406;
190     /**
191      * {@code 407 Proxy Authentication Required}.
192      * @see <a href="http://tools.ietf.org/html/rfc7235#section-3.2">HTTP/1.1: Authentication, section 3.2</a>
193      */
194     int PROXY_AUTHENTICATION_REQUIRED = 407;
195     /**
196      * {@code 408 Request Timeout}.
197      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.7">HTTP/1.1: Semantics and Content, section 6.5.7</a>
198      */
199     int REQUEST_TIMEOUT = 408;
200     /**
201      * {@code 409 Conflict}.
202      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.8">HTTP/1.1: Semantics and Content, section 6.5.8</a>
203      */
204     int CONFLICT = 409;
205     /**
206      * {@code 410 Gone}.
207      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.9">HTTP/1.1: Semantics and Content, section 6.5.9</a>
208      */
209     int GONE = 410;
210     /**
211      * {@code 411 Length Required}.
212      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.10">HTTP/1.1: Semantics and Content, section 6.5.10</a>
213      */
214     int LENGTH_REQUIRED = 411;
215     /**
216      * {@code 412 Precondition failed}.
217      * @see <a href="http://tools.ietf.org/html/rfc7232#section-4.2">HTTP/1.1: Conditional Requests, section 4.2</a>
218      */
219     int PRECONDITION_FAILED = 412;
220     /**
221      * {@code 413 Payload Too Large}.
222      * @since 4.1
223      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.11">HTTP/1.1: Semantics and Content, section 6.5.11</a>
224      */
225     int PAYLOAD_TOO_LARGE = 413;
226     /**
227      * {@code 413 Request Entity Too Large}.
228      * @see <a href="http://tools.ietf.org/html/rfc2616#section-10.4.14">HTTP/1.1, section 10.4.14</a>
229      * @deprecated in favor of {@link #PAYLOAD_TOO_LARGE} which will be returned from {@code HttpStatus.valueOf(413)}
230      */
231     @Deprecated
232     int REQUEST_ENTITY_TOO_LARGE = 413;
233     /**
234      * {@code 414 URI Too Long}.
235      * @since 4.1
236      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.12">HTTP/1.1: Semantics and Content, section 6.5.12</a>
237      */
238     int URI_TOO_LONG = 414;
239     /**
240      * {@code 414 Request-URI Too Long}.
241      * @see <a href="http://tools.ietf.org/html/rfc2616#section-10.4.15">HTTP/1.1, section 10.4.15</a>
242      * @deprecated in favor of {@link #URI_TOO_LONG} which will be returned from {@code HttpStatus.valueOf(414)}
243      */
244     @Deprecated
245     int REQUEST_URI_TOO_LONG = 414;

247      * {@code 415 Unsupported Media Type}.
248      * @see <a href="http://tools.ietf.org/html/rfc7231#section-6.5.13">HTTP/1.1: Semantics and Content, section 6.5.13</a>
249      */
250     int UNSUPPORTED_MEDIA_TYPE = 415;

263      * @see <a href="http://tools.ietf.org/html/rfc2324#section-2.3.2">HTCPCP/1.0</a>
