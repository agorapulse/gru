33  /**
34   * Squad is pool of minions engaged in a test.
35   */
36  public class Squad {
37  
38      private final Map<Class<? extends Minion>, Minion> minions = new LinkedHashMap<>();
39      private final Set<Minion> sorted = new TreeSet<>(Minion.COMPARATOR);
40  
41      /**
42       * Adds a minion to the squad.
43       * @param minion minion who should join the squad
44       */
45      @SuppressWarnings("unchecked")
46      public <M extends Minion> void add(M minion) {
47          minions.put(minion.getClass(), minion);
48  
49          Class<?> sc = minion.getClass().getSuperclass();
50          while (sc != null && !Modifier.isAbstract(sc.getModifiers()) && !sc.equals(Minion.class)
51          ) {
52              minions.put((Class<? extends Minion>) sc, minion);
53              sc = sc.getSuperclass();
54          }
55  
56          sorted.add(minion);
57      }
58  
59      /**
60       * Command minion of given type.
61       *
62       * If the minion is not yet present in the squad it is instantiated using default constructor.
63       * @param minionType type of the minion being commanded
64       * @param command closure executed within context of selected minion
65       */
66      public <M extends Minion> void command(Class<M> minionType, Command<M> command) {
67          command.execute(findOrCreateMinionByType(minionType));
68      }
69  
70      @SuppressWarnings("unchecked")
71      private <M extends Minion> M findOrCreateMinionByType(Class<M> minionType) {
72          M minion = (M) minions.get(minionType);
73  
74          if (minion == null) {
75              try {
76                  minion = minionType.newInstance();
77                  add(minion);
78              } catch (Exception e) {
79                  throw new IllegalStateException("Unable to crete new instance of " + minionType + ". Please, provide public default constructor for the class or add the minion to the squad manually using Gru.engage(minion) method.", e);
80              }
81          }
82          return minion;
83      }
84  
85      /**
86       * Asks minion of given type for something.
87       * @param minionType type of the minion being asked
88       * @param query function executed with the selected minion which returns the result of this method
89       * @return result returned from the query closure or null if minion of given type is not present in the squad
90       */
91      @SuppressWarnings("unchecked")
92      public <T, M extends Minion> T ask(
93          Class<M> minionType,
94          Function<M, T> query
95      ) {
96          M minion = (M) minions.get(minionType);
97  
98          if (minion == null) {
99              return null;
100         }
101 
102         return query.apply(minion);
103     }

121     void verify(Client client, GruContext context) throws Throwable {
122         List<Throwable> errors = new ArrayList<>();
123 
124         for (Minion minion : sorted) {
125             try {
126                 minion.verify(client, this, context);
127             } catch (Throwable t) {
128                 errors.add(t);
129             }
130         }
131 
132         if (errors.size() == 1) {
133             throw errors.get(0);
134         }
135 
136         if (errors.size() > 0) {
137             throw new GroovyAssertAwareMultipleFailureException(errors);
138         }
139     }
