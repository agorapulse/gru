18  package com.agorapulse.gru.agp;
19  
20  import com.agorapulse.gru.Client;
21  import com.agorapulse.gru.MultipartDefinition;
22  import groovy.json.JsonOutput;
23  import groovy.json.JsonSlurper;
24  
25  import java.util.Collections;
26  import java.util.LinkedHashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  class ApiGatewayProxyRequest implements Client.Request {
31  
32      private String body;
33      private Map<String, String> headers;
34      private String httpMethod;
35      private Map<String, String> queryStringParameters;
36      private Map<String, String> pathParameters;
37  
38      private MockContext context = new MockContext();
39  
40      // TODO: stage variables
41      // TODO: request context
42      // TODO: base 64 encoding
43      // TODO: resource
44  
45      private String baseUri;
46      private String uri;
47  
48      @Override
49      public String getBaseUri() {
50          return baseUri;
51      }
52  
53      @Override
54      public void setBaseUri(String baseUri) {
55          this.baseUri = baseUri;
56      }
57  
58      @Override
59      public String getUri() {
60          return uri;
61      }
62  
63      @Override
64      public void setUri(String uri) {
65          this.uri = uri;
66      }
67  
68      @Override
69      public String getMethod() {
70          return httpMethod;
71      }
72  
73      @Override
74      public void setMethod(String method) {
75          httpMethod = method;
76      }
77  
78      @Override
79      public void addHeader(String name, String value) {
80          if (headers == null) {
81              headers = new LinkedHashMap<>();
82          }
83          headers.put(name, value);
84      }
85  
86      @Override
87      public void setJson(String jsonText) {
88          body = jsonText;
89      }
90  
91      @Override
92      public void setContent(String contentType, byte[] payload) {
93          addHeader("Content-Type", contentType);
94          body = new String(payload);
95      }
96  
97      @Override
98      public void addParameter(String name, Object value) {
99          if (queryStringParameters == null) {
100             queryStringParameters = new LinkedHashMap<>();
101         }
102         queryStringParameters.put(name, value == null ? null : value.toString());
103     }
104 
105     @Override
106     public void setMultipart(MultipartDefinition definition) {
107         throw new UnsupportedOperationException("File uploads are currently not supported for API Gateway!");
108     }
109 
110     public void setPathParameters(Map<String, String> pathParameters) {
111         this.pathParameters = Collections.unmodifiableMap(pathParameters);
112     }
113 
114     String getPath() {
115         if (baseUri == null) {
116             return uri;
117         }
118         return (baseUri + uri).replace("//", "/");
119     }
120 
121     public MockContext getContext() {
122         return this.context;
123     }
124 
125     public String toJson() {
126         Map<String, Object> output = new LinkedHashMap<>();
127 
128         output.put("path", getPath());
129 
130         if (body != null) {
131             output.put("body", body);
132         }
133 
134         if (headers != null) {
135             output.put("headers", headers);
136             Map<String, List<String>> multiValueHeaders = new LinkedHashMap<>();
137             headers.forEach((k, v) -> multiValueHeaders.put(k, Collections.singletonList(v)));
138             output.put("multiValueHeaders", multiValueHeaders);
139 
140         }
141 
142         if (httpMethod != null) {
143             output.put("httpMethod", httpMethod);
144 
145         }
146 
147         if (queryStringParameters != null) {
148             output.put("queryStringParameters", queryStringParameters);
149             Map<String, List<String>> multiValueQueryStringParameters = new LinkedHashMap<>();
150             queryStringParameters.forEach((k, v) -> multiValueQueryStringParameters.put(k, Collections.singletonList(v)));
151             output.put("multiValueQueryStringParameters", multiValueQueryStringParameters);
152         }
153 
154         if (pathParameters != null) {
155             output.put("pathParameters", pathParameters);
156         }
157 
158         return JsonOutput.toJson(output);
159     }
160 
161     public String toJson(ApiGatewayConfiguration.MappingConfiguration configuration) {
162         final Map<String, Object> output = new LinkedHashMap<>();
163 
164         if (pathParameters != null && !configuration.getPathParameters().isEmpty()) {
165             configuration.getPathParameters().forEach(parameter -> output.put(parameter, pathParameters.get(parameter)));
166         }
167 
168         if (queryStringParameters != null && !configuration.getQueryStringParameters().isEmpty()) {
169             configuration.getQueryStringParameters().forEach(parameter -> output.put(parameter, queryStringParameters.get(parameter)));
170         }
171 
172         if (body != null) {
173             output.putAll((Map) new JsonSlurper().parseText(body));
174         }
175 
176         return JsonOutput.toJson(output);
177     }
178 }
