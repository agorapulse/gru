18  package com.agorapulse.gru.micronaut;
19  
20  import com.agorapulse.gru.Gru;
21  import com.agorapulse.gru.http.Http;
22  import com.agorapulse.gru.micronaut.okhttp3.MicronautGruConfiguration;
23  import io.micronaut.context.ApplicationContext;
24  import io.micronaut.context.annotation.Bean;
25  import io.micronaut.context.annotation.Factory;
26  import io.micronaut.context.annotation.Requires;
27  
28  import javax.inject.Singleton;
29  import java.util.List;
30  
31  /**
32   * Factory which creates Gru instance for the classes annotated with @MicronautTest.
33   */
34  @Factory
35  @Requires(property = GruFactory.TEST_CLASS_PROPERTY_NAME)
36  public class GruFactory {
37  
38      public static final String TEST_CLASS_PROPERTY_NAME = "micronaut.test.active.spec.clazz";
39  
40      @Singleton
41      @Bean(preDestroy = "close")
42      @SuppressWarnings({"rawtypes", "unchecked"})
43      Gru gru(ApplicationContext context, List<MicronautGruConfiguration> httpConfiguration) {
44          Class testClass = context.getRequiredProperty(TEST_CLASS_PROPERTY_NAME, Class.class);
45          return Gru.create(
46              Micronaut.createLazy(unitTest ->
47                  Http.create(
48                      unitTest,
49                      builder -> httpConfiguration.forEach(conf -> conf.accept(builder))
50                  ),
51                  () -> context.getBean(testClass),
52                  () -> context
53              )
54          );
55      }
56  
57      @Singleton
58      @Bean(preDestroy = "close")
59      @Requires(classes = com.agorapulse.gru.kotlin.Gru.class)
60      com.agorapulse.gru.kotlin.Gru kotlinGru(Gru gru) {
61          return new com.agorapulse.gru.kotlin.Gru(gru);
62      }
63  
64  }
